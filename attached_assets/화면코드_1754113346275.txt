scwin.dataInit = function(subst_nm, subst_cd, mtr_no, dl_cd, dl_nm, subst_pwr, subst_capa, g_subst_capa, mtr_pwr, g_mtr_capa, mtr_capa, dl_pwr, g_dl_capa, dl_capa){
	let subst_vol1_dsc_1_txt = (subst_capa - subst_pwr);
	let subst_vol1_dsc_2_txt = (subst_capa - g_subst_capa);
	let mtr_vol2_dsc_1_txt = (mtr_capa - mtr_pwr);
	let mtr_vol2_dsc_2_txt = (mtr_capa - g_mtr_capa);
	let dl_vol3_dsc_1_txt = (dl_capa - dl_pwr);
	let dl_vol3_dsc_2_txt = (dl_capa - g_dl_capa);

	if(subst_vol1_dsc_1_txt < 0){
		subst_vol1_dsc_1_txt = "0";
	}
	if(subst_vol1_dsc_2_txt < 0){
		subst_vol1_dsc_2_txt = "0";
	}
	if(mtr_vol2_dsc_1_txt < 0){
		mtr_vol2_dsc_1_txt = "0";
	}
	if(mtr_vol2_dsc_2_txt < 0){
		mtr_vol2_dsc_2_txt = "0";
	}
	if(dl_vol3_dsc_1_txt < 0){
		dl_vol3_dsc_1_txt = "0";
	}
	if(dl_vol3_dsc_2_txt < 0){
		dl_vol3_dsc_2_txt = "0";
	}

	txt_subst_nm_dsc.setValue(subst_nm);
	txt_subst_nm_dsc2.setValue(subst_nm);
	txt_subst_nm_dsc3.setValue(subst_nm);
	txt_mtr_no_dsc.setValue("#"+mtr_no);
	txt_mtr_no_dsc2.setValue("#"+mtr_no);
	txt_dl_nm_dsc.setValue(dl_nm);

	txt_subst_nm_title.setValue(subst_nm);
	txt_subst_nm_label.setValue(subst_nm);
	txt_mtr_no_label.setValue("#"+mtr_no);
	txt_dl_nm_label.setValue(dl_nm);

	txt_subst_capa_dsc.setValue(subst_capa);
	txt_g_subst_capa_dsc.setValue(g_subst_capa);

	txt_subst_pwr_dsc.setValue(subst_pwr);
	txt_mtr_capa_dsc.setValue(mtr_capa);
	txt_g_mtr_capa_dsc.setValue(g_mtr_capa);
	txt_mtr_pwr_dsc.setValue(mtr_pwr);
	txt_dl_capa_dsc.setValue(dl_capa);
	txt_g_dl_capa_dsc.setValue(g_dl_capa);
	txt_dl_pwr_dsc.setValue(dl_pwr);

	txt_subst_vol1_dsc_1.setValue(subst_vol1_dsc_1_txt);
	txt_subst_vol1_dsc_2.setValue(subst_vol1_dsc_2_txt);
	txt_mtr_vol2_dsc_1.setValue(mtr_vol2_dsc_1_txt);
	txt_mtr_vol2_dsc_2.setValue(mtr_vol2_dsc_2_txt);
	txt_dl_vol3_dsc_1.setValue(dl_vol3_dsc_1_txt);
	txt_dl_vol3_dsc_2.setValue(dl_vol3_dsc_2_txt);

	if (subst_vol1_dsc_1_txt <= 0 || subst_vol1_dsc_2_txt <= 0) {
		txt_substYn.setValue("변전소 : 여유용량 없음");
		txt_substYn.setStyle("color", "#ff0000");
		//.css('color','#ff0000');
	} else {
		txt_substYn.setValue("변전소 : 접속가능")
		txt_substYn.setStyle("color", "#1479c7");
		//.css('color','#1479c7');
	}
	
	if (mtr_vol2_dsc_1_txt <= 0 || mtr_vol2_dsc_2_txt <= 0) {
		txt_mtrYn.setValue("주변압기 : 여유용량 없음");
		txt_mtrYn.setStyle("color", "#ff0000");
		//.css('color','#ff0000');
	} else {
		txt_mtrYn.setValue("주변압기 : 접속가능");
		txt_mtrYn.setStyle("color", "#1479c7");
		//.css('color','#1479c7');
	}
	
	if (dl_vol3_dsc_1_txt <= 0 || dl_vol3_dsc_2_txt <= 0) {
		txt_dlYn.setValue("배전선로 : 여유용량 없음");
		txt_dlYn.setStyle("color","#ff0000");
		//.css('color','#ff0000');
	} else {
		txt_dlYn.setValue("배전선로 : 접속가능");
		txt_dlYn.setStyle("color","#1479c7");
		//.css('color','#1479c7');
	}
}
